//操作给定的二叉树，将其变换为源二叉树的镜像。

/*

			val(x), left(NULL), right(NULL) {
	}

class Solution {
public:
    
    void Mirror(TreeNode *pRoot) {
        if(pRoot==nullptr)
            return;
        if(pRoot->right==nullptr&&pRoot->left==nullptr)
            return;
        
        
        TreeNode* pTemp = nullptr;
        pTemp = pRoot->left;
        pRoot->left = pRoot->right;
        pRoot->right = pTemp;
        
        if(pRoot->left)
            Mirror(pRoot->left);
        if(pRoot->right)
            Mirror(pRoot->right);
        
        
        
    }
    
    
    
};
